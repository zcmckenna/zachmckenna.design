Robotic Design State


Recently for TSA, I built a two wheeled robot to complete the obstacle course. I had worm gear motors to allow the motors to fit into a small space, a plow to push the objects in the course, and a lever to push the button at the end.

#include <Arduino.h>

#include <PS4BT.h> // Invlude PS4 bluetooth library
#include <usbhub.h> // Include USB library

#include <Wire.h> // Include wire library
#include <Adafruit_MotorShield.h>
#include "utility/Adafruit_MS_PWMServoDriver.h"

#include <Servo.h>

#ifdef dobogusinclude
#include <spi4teensy3.h>
#endif
#include <SPI.h>

USB Usb; // Set up usb shield
BTD Btd(&Usb);
PS4BT PS4(&Btd, PAIR);

Adafruit_MotorShield MS = Adafruit_MotorShield(); // Initialize motor shield and motors
Adafruit_DCMotor *mtr1 = MS.getMotor(1);
Adafruit_DCMotor *mtr4 = MS.getMotor(4);

int firstRun = 1;

int leverAngle = 0;

int presserAngle = 90;

int plowAngle = 0;

int driveMode = 1;

Servo frontLever;
Servo buttonPresser;
Servo plow;

void setup()
{
  Serial.begin(115200);
  if (Usb.Init() == -1)
  {
    Serial.print(F("\nOSC did not start"));
    while (1); // Halt
  }
  Serial.print(F("\nPS4 Bluetooth Library Started"));
  MS.begin(); // Start motor shiekd
  Serial.print(F("\nMotor Shield Started"));
  frontLever.attach(6);
  buttonPresser.attach(5);
  plow.attach(3);
  plow.write(0);
}

void loop()
{
  Usb.Task(); // Start USB Host shield

  if (PS4.connected())
  { // If a PS4 remote is detected
    if (firstRun == 1)
    {
      PS4.setLed(Red);
      firstRun = 0;
    }
    if (driveMode == 1) {
      if (PS4.getAnalogHat(LeftHatY) < 124)
      {
        mtr1->setSpeed(map(PS4.getAnalogHat(LeftHatY), 0, 123, 255, 0));
        mtr1->run(FORWARD);
      }
      else if (PS4.getAnalogHat(LeftHatY) > 129)
      {
        mtr1->setSpeed(map(PS4.getAnalogHat(LeftHatY), 129, 255, 0, 255));
        mtr1->run(BACKWARD);
      }
      else
      {
        mtr1->run(RELEASE);
      }

      if (PS4.getAnalogHat(RightHatY) < 124)
      {
        mtr4->setSpeed(map(PS4.getAnalogHat(RightHatY), 0, 123, 255, 0));
        mtr4->run(FORWARD);
      }
      else if (PS4.getAnalogHat(RightHatY) > 129)
      {
        mtr4->setSpeed(map(PS4.getAnalogHat(RightHatY), 129, 255, 0, 255));
        mtr4->run(BACKWARD);
      }
      else
      {
        mtr4->run(RELEASE);
      }
    }
    if (driveMode == -1) {
      if (PS4.getAnalogHat(LeftHatY) < 124)
      {
        //        mtr1->setSpeed(map(PS4.getAnalogHat(LeftHatY), 0, 123, 255, 0));
        //        mtr1->run(FORWARD);
        mtr4->setSpeed(map(PS4.getAnalogHat(LeftHatY), 0, 123, 255, 0));
        mtr4->run(BACKWARD);
      }
      else if (PS4.getAnalogHat(LeftHatY) > 129)
      {
        //        mtr1->setSpeed(map(PS4.getAnalogHat(LeftHatY), 129, 255, 0, 255));
        //        mtr1->run(BACKWARD);

        mtr4->setSpeed(map(PS4.getAnalogHat(LeftHatY), 129, 255, 0, 255));
        mtr4->run(FORWARD);
      }
      else
      {
        mtr4->run(RELEASE);
      }

      if (PS4.getAnalogHat(RightHatY) < 124)
      {
        //        mtr4->setSpeed(map(PS4.getAnalogHat(RightHatY), 0, 123, 255, 0));
        //        mtr4->run(FORWARD);

        mtr1->setSpeed(map(PS4.getAnalogHat(RightHatY), 0, 123, 255, 0));
        mtr1->run(BACKWARD);
      }
      else if (PS4.getAnalogHat(RightHatY) > 129)
      {
        //        mtr4->setSpeed(map(PS4.getAnalogHat(RightHatY), 129, 255, 0, 255));
        //        mtr4->run(BACKWARD);

        mtr1->setSpeed(map(PS4.getAnalogHat(RightHatY), 129, 255, 0, 255));
        mtr1->run(FORWARD);
      }
      else
      {
        mtr1->run(RELEASE);
      }

    }

    if (PS4.getButtonClick(PS))
    {
      PS4.disconnect();
    }

    if (PS4.getButtonClick(L1))
    {
      presserAngle += 10;
      if (presserAngle > 180) {
        presserAngle = 180;
      }
      buttonPresser.write(presserAngle);
    }

    if (PS4.getButtonClick(R1))
    {
      presserAngle -= 10;
      if (presserAngle < 0) {
        presserAngle = 0;
      }
      buttonPresser.write(presserAngle);
    }

    if (PS4.getButtonClick(UP))
    {
      plowAngle += 10;
      if (plowAngle > 90) {
        plowAngle = 90;
      }
      plow.write(plowAngle);
    }

    if (PS4.getButtonClick(DOWN))
    {
      plowAngle -= 10;
      if (plowAngle < 0) {
        plowAngle = 0;
      }
      plow.write(plowAngle);
    }

    if (PS4.isTouching(0) == true && PS4.getX() < 900) {
      leverAngle += 1;
      if (leverAngle == -1) {
        leverAngle += 1;
      } else if (leverAngle == 180) {
        leverAngle = leverAngle - 1;
      }
      frontLever.write(leverAngle);
      delay(10);
    }

    if (PS4.isTouching(0) == true && PS4.getX() > 900) {
      leverAngle = leverAngle - 1;
      if (leverAngle == -1) {
        leverAngle += 1;
      } else if (leverAngle == 180) {
        leverAngle = leverAngle - 1;
      }
      frontLever.write(leverAngle);
      delay(10);
    }


    if (PS4.getButtonClick(RIGHT)) {
      driveMode = -1;
      PS4.setLed(Green);
    }

    if (PS4.getButtonClick(LEFT)) {
      driveMode = 1;
      PS4.setLed(Yellow);
    }
  }
}
